import { useState } from 'react';
import './App.css';

function App() {
  const [messageInput, setMessageInput] = useState('');
  const [displayedMessage, setDisplayedMessage] = useState('Hello World');

  function displayMessage() {
    setDisplayedMessage(messageInput);
  }

  return (
    <div className="container">
      <h1>React Message App</h1>

      <label htmlFor="messageInput">Message to Display</label>
      <input
        id="messageInput"
        type="text"
        value={messageInput}
        onChange={(e) => setMessageInput(e.target.value)}
        placeholder="Enter your message"
      />

      <button onClick={displayMessage}>Display Message</button>

      <h2>{displayedMessage}</h2>
    </div>
  );
}

export default App;
